    &m = 0 \\
    &p-q=1 \\
\end{cases}

# 习题二


证明: 

$$
g(x)|f_1(x)+2f_2(x), g(x)|3f_1(x)-4f_2(x), 则g(x)|f_1(x), g(x)|f_2(x)
$$

即有

$$
f_1(x)+2f_x(x)=g(x)h_1(x), 3f_1(x)-4f_x(x)=g(x)h_2(x)
$$

所以

$$
f_1(x)=\frac{2}{5}f_1(x)+\frac{1}{5}f_2(x)
$$

$$
f_2(x)=\frac{3}{10}f_1(x)-\frac{1}{10}f_2(x)
$$

所以

$$
g(x)|f_1(x), g(x)|f_2(x)
$$
