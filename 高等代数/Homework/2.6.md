\end{vmatrix}_{n}

2\cos \alpha\cos[(n-1)\alpha]-

0 &0 &0 &\cdots &2\cos \alpha &0 \\

\end{vmatrix}_{n-1}
\\ =&
2\cos \alpha\cos[(n-1)\alpha]-
\begin{vmatrix}
\cos \alpha &1 &0 &\cdots &0 \\
1 &2\cos \alpha &1 &\cdots &0 \\

\vdots &\vdots &\vdots & &\vdots \\
